{
    //简单变量声明
    int a;
    //指针声明
    int *b;
    //数组声明
    char[10] str;
    //高位数组声明
    float[5][5] f;
    //过程声明
    int print(){}
    //变量赋值
    a = 5;
    //数组赋值
    str[1] = 20;
    //高维数组赋值
    f[1][2] = 3;
    //分支，逻辑表达式
    if(a>3)
        a = 4;
    else
        a = 3;
    //循环
    do
        //算术表达式
        a = a + 1;
    while(a<10);
    //过程调用
    print(a);
}
